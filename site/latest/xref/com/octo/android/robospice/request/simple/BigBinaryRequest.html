14  /**
15   * Downloads big images in size. All data is passed to the listener using file
16   * system. This class is meant to help download big images. If you wish to
17   * download smaller documents, you would be better using
18   * {@link SmallBinaryRequest}.
19   * @author sni & jva
20   */
21  public class BigBinaryRequest extends BinaryRequest {
22  
23      protected File cacheFile;
24  
25      public BigBinaryRequest(final String url, final File cacheFile) {
26          super(url);
27          this.cacheFile = cacheFile;
28      }
29  
30      @Override
31      public InputStream processStream(final int contentLength,
32          final InputStream inputStream) throws IOException {
33          // touch
34          boolean isTouchedNow = cacheFile.setLastModified(System
35              .currentTimeMillis());
36          if (!isTouchedNow) {
37              Ln.d("Modification time of file %s could not be changed normally ",
38                  cacheFile.getAbsolutePath());
39          }
40          final OutputStream fileOutputStream = new FileOutputStream(cacheFile);
41          readBytes(inputStream, new ProgressByteProcessor(fileOutputStream,
42              contentLength));
43          IOUtils.closeQuietly(fileOutputStream);
44          return new FileInputStream(cacheFile);
45      }
46  
47      public File getCacheFile() {
48          return cacheFile;
49      }
