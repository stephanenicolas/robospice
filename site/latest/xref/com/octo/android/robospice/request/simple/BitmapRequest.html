83      @Override
84      public Bitmap loadDataFromNetwork() throws Exception {
85          try {
86              final HttpURLConnection httpURLConnection = (HttpURLConnection) new URL(url).openConnection();
87              processStream(httpURLConnection.getContentLength(), httpURLConnection.getInputStream());
88  
89              if (width != -1 && height != -1) {
90                  this.options = new BitmapFactory.Options();
91                  options.inJustDecodeBounds = true;
92                  BitmapFactory.decodeFile(cacheFile.getAbsolutePath(), options);
93                  options.inSampleSize = calculateInSampleSize(options, width, height);
94                  options.inJustDecodeBounds = false;
95                  options.inPurgeable = true;
96                  return BitmapFactory.decodeFile(cacheFile.getAbsolutePath(), options);
97              } else {
98                  return BitmapFactory.decodeFile(cacheFile.getAbsolutePath(), options);
99              }
100         } catch (final MalformedURLException e) {
101             Ln.e(e, "Unable to create URL");
102             return null;
103         } catch (final IOException e) {
104             Ln.e(e, "Unable to download binary");
105             return null;
106         }

117     public void processStream(int contentLength, final InputStream inputStream) throws IOException {
118         OutputStream fileOutputStream = null;
119         try {
120             // touch
121             boolean isTouchedNow = cacheFile.setLastModified(System.currentTimeMillis());
122             if (!isTouchedNow) {
123                 Ln.d("Modification time of file %s could not be changed normally ", cacheFile.getAbsolutePath());
124             }
125             fileOutputStream = new FileOutputStream(cacheFile);
126             readBytes(inputStream, new ProgressByteProcessor(this, fileOutputStream, contentLength));
127         } finally {
128             IOUtils.closeQuietly(fileOutputStream);
129         }
130     }

132     /**
133      * Inspired from Guava com.google.common.io.ByteStreams
134      */
135     protected void readBytes(final InputStream in, final ProgressByteProcessor processor) throws IOException {
136         final byte[] buf = new byte[BUF_SIZE];
137         try {
138             int amt;
139             do {
140                 amt = in.read(buf);
141                 if (amt == -1) {
142                     break;
143                 }
144             } while (processor.processBytes(buf, 0, amt));
145         } finally {
146             IOUtils.closeQuietly(in);
147         }
148     }

150     private static int calculateInSampleSize(BitmapFactory.Options options, int reqWidth, int reqHeight) {
151         // Raw height and width of image
152         final int height = options.outHeight;
153         final int width = options.outWidth;
154         int inSampleSize = 1;
155 
156         if (height > reqHeight || width > reqWidth) {
157             if (width > height) {
158                 inSampleSize = Math.round((float) height / (float) reqHeight);
159             } else {
160                 inSampleSize = Math.round((float) width / (float) reqWidth);
161             }
162         }
163         return inSampleSize;
164     }
