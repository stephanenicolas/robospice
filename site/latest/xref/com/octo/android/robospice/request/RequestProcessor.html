71      public void addRequest(final CachedSpiceRequest<?> request, final Set<RequestListener<?>> listRequestListener) {
72          if (isStopped) {
73              Ln.d("Dropping request : " + request + " as processor is stopped.");
74              return;
75          }
76  
77          Ln.d("Adding request to queue " + hashCode() + ": " + request + " size is " + mapRequestToRequestListener.size());
78  
79          if (request.isCancelled()) {
80              synchronized (mapRequestToRequestListener) {
81                  for (final CachedSpiceRequest<?> cachedSpiceRequest : mapRequestToRequestListener.keySet()) {
82                      if (cachedSpiceRequest.equals(request)) {
83                          cachedSpiceRequest.cancel();
84                          requestProgressManager.notifyListenersOfRequestCancellation(request);
85                          return;
86                      }
87                  }
88              }
89          }
90  
91          boolean aggregated = false;
92          Set<RequestListener<?>> listRequestListenerForThisRequest = mapRequestToRequestListener.get(request);
93  
94          if (listRequestListenerForThisRequest == null) {
95              if (request.isProcessable()) {
96                  Ln.d("Adding entry for type %s and cacheKey %s.", request.getResultType(), request.getRequestCacheKey());
97                  listRequestListenerForThisRequest = Collections.synchronizedSet(new HashSet<RequestListener<?>>());
98                  this.mapRequestToRequestListener.put(request, listRequestListenerForThisRequest);
99              }
100         } else {
101             Ln.d("Request for type %s and cacheKey %s already exists.", request.getResultType(), request.getRequestCacheKey());
102             aggregated = true;
103         }
104 
105         if (listRequestListener != null && listRequestListenerForThisRequest != null) {
106             listRequestListenerForThisRequest.addAll(listRequestListener);
107         }
108 
109         if (aggregated) {
110             requestProgressManager.notifyListenersOfRequestAggregated(request, listRequestListener);
111             return;
112         }
113 
114         if (request.isProcessable()) {
115             requestProgressManager.notifyListenersOfRequestAdded(request, listRequestListener);
116         } else {
117             if (listRequestListenerForThisRequest == null) {
118                 requestProgressManager.notifyListenersOfRequestNotFound(request, listRequestListener);
119             }
120             requestProgressManager.notifyOfRequestProcessed(request, listRequestListener);
121             // we have to return if request is not processable.
122             // fix bug https://github.com/octo-online/robospice/issues/215
123             return;
124         }
125 
126         final RequestCancellationListener requestCancellationListener = new RequestCancellationListener() {
127 
128             @Override
129             public void onRequestCancelled() {
130                 requestProgressManager.notifyListenersOfRequestCancellation(request);
131                 mapRequestToRequestListener.remove(request);
132             }
133         };
134         request.setRequestCancellationListener(requestCancellationListener);
135 
136         if (request.isCancelled()) {
137             requestProgressManager.notifyListenersOfRequestCancellation(request);
138             mapRequestToRequestListener.remove(request);
139             return;
140         } else {
141             requestRunner.executeRequest(request);
142         }
143     }
