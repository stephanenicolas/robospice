70      public void addRequest(final CachedSpiceRequest<?> request, final Set<RequestListener<?>> listRequestListener) {
71          Ln.d("Adding request to queue " + hashCode() + ": " + request + " size is " + mapRequestToRequestListener.size());
72  
73          if (request.isCancelled()) {
74              synchronized (mapRequestToRequestListener) {
75                  for (final CachedSpiceRequest<?> cachedSpiceRequest : mapRequestToRequestListener.keySet()) {
76                      if (cachedSpiceRequest.equals(request)) {
77                          cachedSpiceRequest.cancel();
78                          requestProgressManager.notifyListenersOfRequestCancellation(request, null);
79                          return;
80                      }
81                  }
82              }
83          }
84  
85          boolean aggregated = false;
86          Set<RequestListener<?>> listRequestListenerForThisRequest = mapRequestToRequestListener.get(request);
87  
88          if (listRequestListenerForThisRequest == null) {
89              if (request.isProcessable()) {
90                  Ln.d("Adding entry for type %s and cacheKey %s.", request.getResultType(), request.getRequestCacheKey());
91                  listRequestListenerForThisRequest = Collections.synchronizedSet(new HashSet<RequestListener<?>>());
92                  this.mapRequestToRequestListener.put(request, listRequestListenerForThisRequest);
93              }
94          } else {
95              Ln.d("Request for type %s and cacheKey %s already exists.", request.getResultType(), request.getRequestCacheKey());
96              aggregated = true;
97          }
98  
99          if (listRequestListener != null && listRequestListenerForThisRequest != null) {
100             listRequestListenerForThisRequest.addAll(listRequestListener);
101         }
102 
103         if (aggregated) {
104             requestProgressManager.notifyListenersOfRequestAggregated(request, listRequestListener);
105             return;
106         }
107 
108         if (request.isProcessable()) {
109             requestProgressManager.notifyListenersOfRequestAdded(request, listRequestListener);
110         } else {
111             if (listRequestListenerForThisRequest == null) {
112                 requestProgressManager.notifyListenersOfRequestNotFound(request, listRequestListener);
113             }
114             requestProgressManager.notifyOfRequestProcessed(request, listRequestListener);
115         }
116 
117         final RequestCancellationListener requestCancellationListener = new RequestCancellationListener() {
118 
119             @Override
120             public void onRequestCancelled() {
121                 mapRequestToRequestListener.remove(request);
122                 requestProgressManager.notifyListenersOfRequestCancellation(request, listRequestListener);
123             }
124         };
125         request.setRequestCancellationListener(requestCancellationListener);
126 
127         if (request.isCancelled()) {
128             mapRequestToRequestListener.remove(request);
129             requestProgressManager.notifyListenersOfRequestCancellation(request, listRequestListener);
130             return;
131         } else {
132             requestRunner.executeRequest(request);
133         }
134     }
