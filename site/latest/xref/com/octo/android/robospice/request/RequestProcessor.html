108     public void addRequest(final CachedSpiceRequest<?> request, final Set<RequestListener<?>> listRequestListener) {
109         Ln.d("Adding request to queue " + hashCode() + ": " + request + " size is " + mapRequestToRequestListener.size());
110 
111         if (request.isCancelled()) {
112             synchronized (mapRequestToRequestListener) {
113                 for (final CachedSpiceRequest<?> cachedSpiceRequest : mapRequestToRequestListener.keySet()) {
114                     if (cachedSpiceRequest.equals(request)) {
115                         cachedSpiceRequest.cancel();
116                         return;
117                     }
118                 }
119             }
120         }
121 
122         boolean aggregated = false;
123         if (listRequestListener != null) {
124             Set<RequestListener<?>> listRequestListenerForThisRequest = mapRequestToRequestListener.get(request);
125 
126             if (listRequestListenerForThisRequest == null) {
127                 if (request.isProcessable()) {
128                     Ln.d(String.format("Adding entry for type %s and cacheKey %s.", request.getResultType(), request.getRequestCacheKey()));
129                     listRequestListenerForThisRequest = new HashSet<RequestListener<?>>();
130                     this.mapRequestToRequestListener.put(request, listRequestListenerForThisRequest);
131                 }
132             } else {
133                 Ln.d(String.format("Request for type %s and cacheKey %s already exists.", request.getResultType(), request.getRequestCacheKey()));
134                 aggregated = true;
135             }
136 
137             if (listRequestListenerForThisRequest != null) {
138                 listRequestListenerForThisRequest.addAll(listRequestListener);
139             }
140             if (request.isProcessable()) {
141                 notifyListenersOfRequestProgress(request, listRequestListener, request.getProgress());
142             }
143         }
144 
145         if (aggregated) {
146             return;
147         }
148 
149         final RequestCancellationListener requestCancellationListener = new RequestCancellationListener() {
150 
151             @Override
152             public void onRequestCancelled() {
153                 mapRequestToRequestListener.remove(request);
154                 notifyListenersOfRequestCancellation(request, listRequestListener);
155             }
156         };
157         request.setRequestCancellationListener(requestCancellationListener);
158 
159         if (request.isCancelled()) {
160             mapRequestToRequestListener.remove(request);
161             notifyListenersOfRequestCancellation(request, listRequestListener);
162             return;
163         } else if (!request.isProcessable()) {
164             notifyOfRequestProcessed(request);
165             return;
166         } else {
167             planRequestExecution(request);
168         }
169     }

171     protected <T> void processRequest(final CachedSpiceRequest<T> request) {
172 
173         Ln.d("Processing request : " + request);
174 
175         T result = null;
176 
177         // add a progress listener to the request to be notified of
178         // progress during load data from network
179         final RequestProgressListener requestProgressListener = new RequestProgressListener() {
180             @Override
181             public void onRequestProgressUpdate(final RequestProgress progress) {
182                 final Set<RequestListener<?>> listeners = mapRequestToRequestListener.get(request);
183                 notifyListenersOfRequestProgress(request, listeners, progress);
184             }
185         };
186         request.setRequestProgressListener(requestProgressListener);
187 
188         if (request.getRequestCacheKey() != null && request.getCacheDuration() != DurationInMillis.ALWAYS_EXPIRED) {
189             // First, search data in cache
190             try {
191                 Ln.d("Loading request from cache : " + request);
192                 request.setStatus(RequestStatus.READING_FROM_CACHE);
193                 result = loadDataFromCache(request.getResultType(), request.getRequestCacheKey(), request.getCacheDuration());
194                 // if something is found in cache, fire result and finish
195                 // request
196                 if (result != null) {
197                     Ln.d("Request loaded from cache : " + request + " result=" + result);
198                     notifyListenersOfRequestSuccess(request, result);
199                     return;
200                 } else if (request.isAcceptingDirtyCache()) {
201                     // as a fallback, some request may accept whatever is in the
202                     // cache but still
203                     // want an update from network.
204                     result = loadDataFromCache(request.getResultType(), request.getRequestCacheKey(), DurationInMillis.ALWAYS_RETURNED);
205                     if (result != null) {
206                         notifyListenersOfRequestSuccessButDontCompleteRequest(request, result);
207                     }
208                 }
209             } catch (final SpiceException e) {
210                 Ln.d(e, "Cache file could not be read.");
211                 if (failOnCacheError) {
212                     handleRetry(request, e);
213                     return;
214                 }
215                 cacheManager.removeDataFromCache(request.getResultType(), request.getRequestCacheKey());
216                 Ln.d(e, "Cache file deleted.");
217             }
218         }
219 
220         // if result is not in cache, load data from network
221         Ln.d("Cache content not available or expired or disabled");
222         if (!isNetworkAvailable(applicationContext) && !request.isOffline()) {
223             Ln.e("Network is down.");
224             handleRetry(request, new NoNetworkException());
225             return;
226         }
227 
228         // network is ok, load data from network
229         try {
230             if (request.isCancelled()) {
231                 return;
232             }
233             Ln.d("Calling netwok request.");
234             request.setStatus(RequestStatus.LOADING_FROM_NETWORK);
235             result = request.loadDataFromNetwork();
236             Ln.d("Network request call ended.");
237         } catch (final Exception e) {
238             if (!request.isCancelled()) {
239                 Ln.e(e, "An exception occured during request network execution :" + e.getMessage());
240                 handleRetry(request, new NetworkException("Exception occured during invocation of web service.", e));
241             } else {
242                 Ln.e("An exception occured during request network execution but request was cancelled, so listeners are not called.");
243             }
244             return;
245         }
246 
247         if (result != null && request.getRequestCacheKey() != null) {
248             // request worked and result is not null, save
249             // it to cache
250             try {
251                 if (request.isCancelled()) {
252                     return;
253                 }
254                 Ln.d("Start caching content...");
255                 request.setStatus(RequestStatus.WRITING_TO_CACHE);
256                 result = saveDataToCacheAndReturnData(result, request.getRequestCacheKey());
257                 if (request.isCancelled()) {
258                     return;
259                 }
260                 notifyListenersOfRequestSuccess(request, result);
261                 return;
262             } catch (final SpiceException e) {
263                 Ln.d("An exception occured during service execution :" + e.getMessage(), e);
264                 if (failOnCacheError) {
265                     handleRetry(request, e);
266                     return;
267                 } else {
268                     if (request.isCancelled()) {
269                         return;
270                     }
271                     // result can't be saved to
272                     // cache but we reached that
273                     // point after a success of load
274                     // data from
275                     // network
276                     notifyListenersOfRequestSuccess(request, result);
277                 }
278                 cacheManager.removeDataFromCache(request.getResultType(), request.getRequestCacheKey());
279                 Ln.d(e, "Cache file deleted.");
280             }
281         } else {
282             // result can't be saved to cache but we reached
283             // that point after a success of load data from
284             // network
285             notifyListenersOfRequestSuccess(request, result);
286             return;
287         }
288     }

290     private void planRequestExecution(final CachedSpiceRequest<?> request) {
291         Future<?> future = executorService.submit(new PriorityRunnable() {
292             @Override
293             public void run() {
294                 try {
295                     processRequest(request);
296                 } catch (final Throwable t) {
297                     Ln.d(t, "An unexpected error occured when processsing request %s", request.toString());
298                 }
299             }
300 
301             @Override
302             public int getPriority() {
303                 return request.getPriority();
304             }
305         });
306         request.setFuture(future);
307     }

309     private void handleRetry(final CachedSpiceRequest<?> request, final SpiceException e) {
310         if (request.getRetryPolicy() != null) {
311             request.getRetryPolicy().retry(e);
312             if (request.getRetryPolicy().getRetryCount() > 0) {
313                 new Thread(new Runnable() {
314                     @Override
315                     public void run() {
316                         try {
317                             Thread.sleep(request.getRetryPolicy().getDelayBeforeRetry());
318                             planRequestExecution(request);
319                         } catch (InterruptedException e) {
320                             Ln.e(e, "Retry attempt failed for request " + request);
321                         }
322                     }
323                 }).start();
324                 return;
325             }
326         }
327         notifyListenersOfRequestFailure(request, e);
328     }

446     private static class ProgressRunnable implements Runnable {
447         private final RequestProgress progress;
448         private final Set<RequestListener<?>> listeners;
449 
450         public ProgressRunnable(final Set<RequestListener<?>> listeners, final RequestProgress progress) {
451             this.progress = progress;
452             this.listeners = listeners;
453         }
454 
455         @Override
456         public void run() {
457 
458             if (listeners == null) {
459                 return;
460             }
461 
462             Ln.v("Notifying " + listeners.size() + " listeners of progress " + progress);
463             for (final RequestListener<?> listener : listeners) {
464                 if (listener != null && listener instanceof RequestProgressListener) {
465                     Ln.v("Notifying %s", listener.getClass().getSimpleName());
466                     ((RequestProgressListener) listener).onRequestProgressUpdate(progress);
467                 }
468             }
469         }

472     private static class ResultRunnable<T> implements Runnable {
473 
474         private SpiceException spiceException;
475         private T result;
476         private final Set<RequestListener<?>> listeners;
477 
478         public ResultRunnable(final Set<RequestListener<?>> listeners, final T result) {
479             this.result = result;
480             this.listeners = listeners;
481         }
482 
483         public ResultRunnable(final Set<RequestListener<?>> listeners, final SpiceException spiceException) {
484             this.spiceException = spiceException;
485             this.listeners = listeners;
486         }
487 
488         @Override
489         public void run() {
490             if (listeners == null) {
491                 return;
492             }
493 
494             final String resultMsg = spiceException == null ? "success" : "failure";
495             Ln.v("Notifying " + listeners.size() + " listeners of request " + resultMsg);
496             for (final RequestListener<?> listener : listeners) {
497                 if (listener != null) {
498                     @SuppressWarnings("unchecked")
499                     final RequestListener<T> listenerOfT = (RequestListener<T>) listener;
500                     Ln.v("Notifying %s", listener.getClass().getSimpleName());
501                     if (spiceException == null) {
502                         listenerOfT.onRequestSuccess(result);
503                     } else {
504                         listener.onRequestFailure(spiceException);
505                     }
506                 }
507             }
508         }
509     }

547     protected void notifyOfRequestProcessed(final CachedSpiceRequest<?> request) {
548         Ln.v("Removing %s  size is %d", request, mapRequestToRequestListener.size());
549         mapRequestToRequestListener.remove(request);
