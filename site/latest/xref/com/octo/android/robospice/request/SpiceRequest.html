37      private void checkInnerClassDeclarationToPreventMemoryLeak() {
38          // thanx to Cyril Mottier for this contribution
39          // prevent devs from creating memory leaks by using inner
40          // classes of contexts
41          if (getClass().isMemberClass()
42              && Context.class.isAssignableFrom(getClass().getDeclaringClass())
43              && !Modifier.isStatic(getClass().getModifiers())) {
44              throw new IllegalArgumentException(
45                  "Requests must be either non-inner classes or a static inner member class of Context : "
46                      + getClass());
47          }
48      }
