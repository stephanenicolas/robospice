11  /**
12   * Uses base 64 to sanitize/de-sanitize keys. Only applies to cache keys that
13   * are strings. Keys sanitized by this class can be safely used to create file
14   * names (and urls).
15   * @author SNI
16   */
17  @TargetApi(Build.VERSION_CODES.FROYO)
18  public class DefaultKeySanitizer implements KeySanitizer {
19  
20      // ----------------------------------
21      // CONSTANTS
22      // ----------------------------------
23      private static final int BASE64_FLAGS = Base64.URL_SAFE + Base64.NO_WRAP + Base64.NO_PADDING;
24      private static final String UTF8_CHARSET_NAME = "UTF-8";
25  
26      // ----------------------------------
27      // API
28      // ----------------------------------
29  
30      @Override
31      public Object sanitizeKey(Object cacheKey) throws KeySanitationExcepion {
32          if (!(cacheKey instanceof String)) {
33              throw new KeySanitationExcepion(DefaultKeySanitizer.class.getSimpleName() + " can only be used with Strings cache keys.");
34          }
35          try {
36              return Base64.encodeToString(((String) cacheKey).getBytes(UTF8_CHARSET_NAME), BASE64_FLAGS);
37          } catch (UnsupportedEncodingException e) {
38              throw new KeySanitationExcepion(e);
39          }
40      }
41  
42      @Override
43      public Object desanitizeKey(Object sanitzedCacheKey) throws KeySanitationExcepion {
44          if (!(sanitzedCacheKey instanceof String)) {
45              throw new KeySanitationExcepion(DefaultKeySanitizer.class.getSimpleName() + " can only be used with Strings cache keys.");
46          }
47          try {
48              return new String(Base64.decode((String) sanitzedCacheKey, BASE64_FLAGS), UTF8_CHARSET_NAME);
49          } catch (UnsupportedEncodingException e) {
50              throw new KeySanitationExcepion(e);
51          }
52      }
53  }
