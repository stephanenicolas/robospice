116     private <T> void clearTable(Class<T> clazz) throws SQLException {
117         for (Field field : clazz.getDeclaredFields()) {
118             ForeignCollectionField annotation = field.getAnnotation(ForeignCollectionField.class);
119             if (annotation != null) {
120                 ParameterizedType stringListType = (ParameterizedType) field.getGenericType();
121                 Class<?> itemInListClass = (Class<?>) stringListType.getActualTypeArguments()[0];
122                 clearTable(itemInListClass);
123             }
124         }
125         TableUtils.dropTable(getConnectionSource(), clazz, true);
126         TableUtils.createTableIfNotExists(getConnectionSource(), clazz);
127     }
