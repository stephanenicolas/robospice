41      @Override
42      protected Bitmap readCacheDataFromFile(File file) throws CacheLoadingException {
43          Bitmap data;
44          FileInputStream is = null;
45          try {
46              is = new FileInputStream(file.getAbsolutePath());
47              data = BitmapFactory.decodeStream(is, null, decodingOptions);
48          } catch (Throwable ex) {
49              throw new CacheLoadingException(String.format("Found the file %s but could not decode bitmap.", file.getAbsolutePath()), ex);
50          } finally {
51              if (is != null) {
52                  IOUtils.closeQuietly(is);
53              }
54          }
55          if (data == null) {
56              throw new CacheLoadingException(String.format("Found the file %s but could not decode bitmap.", file.getAbsolutePath()));
57          }
58          return data;
59      }
60  
61      @Override
62      public Bitmap saveDataToCacheAndReturnData(Bitmap data, Object cacheKey) throws CacheSavingException {
63          BufferedOutputStream out = null;
64  
65          try {
66              File cacheFile = getCacheFile(cacheKey);
67              out = new BufferedOutputStream(new FileOutputStream(cacheFile));
68  
69              boolean didCompress = data.compress(compressFormat, quality, out);
70              if (!didCompress) {
71                  throw new CacheSavingException(String.format("Could not compress bitmap for path: %s", getCacheFile(cacheKey).getAbsolutePath()));
72              }
73  
74              return data;
75          } catch (IOException e) {
76              throw new CacheSavingException(e);
77          } finally {
78              IOUtils.closeQuietly(out);
79          }
80      }
