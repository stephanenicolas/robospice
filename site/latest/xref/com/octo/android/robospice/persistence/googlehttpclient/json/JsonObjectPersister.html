48      @Override
49      protected T readCacheDataFromFile(File file) throws CacheLoadingException {
50          try {
51              JsonParser jsonParser = jsonFactory.createJsonParser(new FileReader(file));
52              T result = jsonParser.parse(getHandledClass(), null);
53              jsonParser.close();
54              return result;
55          } catch (FileNotFoundException e) {
56              // Should not occur (we test before if file exists)
57              // Do not throw, file is not cached
58              Ln.w("file " + file.getAbsolutePath() + " does not exists", e);
59              return null;
60          } catch (Exception e) {
61              throw new CacheLoadingException(e);
62          }
63      }

65      @Override
66      public T saveDataToCacheAndReturnData(final T data, final Object cacheKey) throws CacheSavingException {
67  
68          try {
69              if (isAsyncSaveEnabled()) {
70                  Thread t = new Thread() {
71                      @Override
72                      public void run() {
73                          try {
74                              saveData(data, cacheKey);
75                          } catch (IOException e) {
76                              Ln.e(e, "An error occured on saving request " + cacheKey + " data asynchronously");
77                          } catch (CacheSavingException e) {
78                              Ln.e(e, "An error occured on saving request " + cacheKey + " data asynchronously");
79                          }
80                      };
81                  };
82                  t.start();
83              } else {
84                  saveData(data, cacheKey);
85              }
86          } catch (CacheSavingException e) {
87              throw e;
88          } catch (Exception e) {
89              throw new CacheSavingException(e);
90          }
91          return data;
92      }

94      private void saveData(T data, Object cacheKey) throws IOException, CacheSavingException {
95          // transform the content in json to store it in the cache
96          JsonGenerator jsonGenerator = jsonFactory.createJsonGenerator(new FileWriter(getCacheFile(cacheKey)));
97          jsonGenerator.serialize(data);
98          jsonGenerator.flush();
99          jsonGenerator.close();
