42      @Override
43      public T loadDataFromCache(Object cacheKey, long maxTimeInCacheBeforeExpiry) throws CacheLoadingException {
44          CacheItem<T> cacheItem = lruCache.get(cacheKey);
45  
46          if (cacheItem == null) {
47              Ln.d("Miss from lru cache for %s", cacheKey);
48              if (decoratedPersister != null) {
49                  T data = decoratedPersister.loadDataFromCache(cacheKey, maxTimeInCacheBeforeExpiry);
50                  if (data == null) {
51                      return null;
52                  }
53                  CacheItem<T> item = new CacheItem<T>(decoratedPersister.getCreationDateInCache(cacheKey), data);
54                  Ln.d("Put in lru cache after miss");
55                  lruCache.put(cacheKey, item);
56                  return data;
57              }
58              return null;
59          } else {
60              Ln.d("Hit from lru cache for %s", cacheKey);
61              boolean dataCanExpire = maxTimeInCacheBeforeExpiry != DurationInMillis.ALWAYS_RETURNED;
62              boolean dataIsNotExpired = System.currentTimeMillis() - cacheItem.getCreationDate() <= maxTimeInCacheBeforeExpiry;
63              if (!dataCanExpire || dataIsNotExpired) {
64                  return cacheItem.getData();
65              }
66              return null;
67          }
68      }

71      public T saveDataToCacheAndReturnData(T data, Object cacheKey) throws CacheSavingException {
72          CacheItem<T> itemToCache = new CacheItem<T>(data);
73          lruCache.put(cacheKey, itemToCache);
74          Ln.d("Put in lru cache for %s", cacheKey);
75  
76          if (decoratedPersister != null) {
77              decoratedPersister.saveDataToCacheAndReturnData(data, cacheKey);
78          }

80          return data;

83      @Override
84      public boolean isDataInCache(Object cacheKey, long maxTimeInCacheBeforeExpiry) {
85          CacheItem<T> cacheItem = lruCache.get(cacheKey);
86  
87          if (cacheItem == null) {
88              if (decoratedPersister != null) {
89                  return decoratedPersister.isDataInCache(cacheKey, maxTimeInCacheBeforeExpiry);
90              }
91              return false;
92          } else {
93              boolean dataCanExpire = maxTimeInCacheBeforeExpiry != DurationInMillis.ALWAYS_RETURNED;
94              boolean dataIsNotExpired = System.currentTimeMillis() - cacheItem.getCreationDate() <= maxTimeInCacheBeforeExpiry;
95              return !dataCanExpire || dataIsNotExpired;
96          }
97      }

115         if (decoratedPersister != null) {

117         } else {
118             Map<Object, CacheItem<T>> cacheMap = lruCache.snapshot();
119             List<T> allData = new ArrayList<T>();
