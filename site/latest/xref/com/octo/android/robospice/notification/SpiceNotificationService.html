18  /**
19   * Will create notifications to display the progress of a given request. This
20   * class is a base class to create such a service. Implementations will only
21   * focus on creating notifications to "follow" the status of given request. All
22   * informations about the request to track are provided to the service via an
23   * Intent. This services will be automatically stopped when the requests has
24   * been fully processed.
25   * @author SNI

45      public static Intent createIntent(final Context context, final Class<? extends SpiceNotificationService> clazz, final Class<? extends SpiceService> spiceServiceClass, final int notificationId,
46          final Class<?> requestResultType, final String cacheKey, final boolean foreground) {
47          final Intent intent = new Intent(context, clazz);
48          intent.putExtra(BUNDLE_KEY_NOTIFICATION_ID, notificationId);
49          intent.putExtra(BUNDLE_KEY_SERVICE_CLASS, spiceServiceClass);
50          intent.putExtra(BUNDLE_KEY_REQUEST_CLASS, requestResultType);
51          intent.putExtra(BUNDLE_KEY_REQUEST_CACHE_KEY, cacheKey);
52          intent.putExtra(BUNDLE_KEY_FOREGROUND, foreground);
53          return intent;
54      }

61      @SuppressWarnings({ "unchecked", "rawtypes", "deprecation" })
62      @Override
63      public final void onStart(final Intent intent, final int startId) {
64          super.onStart(intent, startId);
65          notificationId = intent.getIntExtra(BUNDLE_KEY_NOTIFICATION_ID, DEFAULT_ROBOSPICE_NOTIFICATION_ID);
66          requestClass = (Class<?>) intent.getSerializableExtra(BUNDLE_KEY_REQUEST_CLASS);
67          requestCacheKey = intent.getStringExtra(BUNDLE_KEY_REQUEST_CACHE_KEY);
68          spiceServiceClass = (Class<? extends SpiceService>) intent.getSerializableExtra(BUNDLE_KEY_SERVICE_CLASS);
69          if (spiceServiceClass == null) {
70              throw new RuntimeException("Please specify a service class to monitor. Use #createIntent as helper.");
71          }
72          foreground = intent.getBooleanExtra(BUNDLE_KEY_FOREGROUND, true);
73  
74          spiceManager = new SpiceManager(spiceServiceClass);
75          notificationManager = (NotificationManager) getSystemService(NOTIFICATION_SERVICE);
76          spiceManager.start(this);
77          spiceManager.addListenerIfPending(requestClass, requestCacheKey, new NotificationRequestListener());
78  
79          if (foreground) {
80              startForeground(notificationId, onCreateForegroundNotification());
81          }
82      }
83  
84      @Override
85      public final void onDestroy() {
86          spiceManager.shouldStop();
87          super.onDestroy();
88      }

104     private class NotificationRequestListener<T> implements RequestListener<T>, RequestProgressListener {
105 
106         @Override
107         public void onRequestFailure(final SpiceException arg0) {
108             final Notification notification = onCreateNotificationForRequestFailure(arg0);
109             notificationManager.notify(notificationId, notification);
110             stopSelf();
111         }
112 
113         @Override
114         public void onRequestSuccess(final T result) {
115             final Notification notification = onCreateNotificationForRequestSuccess();
116             notificationManager.notify(notificationId, notification);
117             stopSelf();
118         }
119 
120         @Override
121         public void onRequestProgressUpdate(final RequestProgress progress) {
122             final Notification notification = onCreateNotificationForRequestProgress(progress);
123             notificationManager.notify(notificationId, notification);
124 
125             if (progress.getStatus() == RequestStatus.COMPLETE) {
126                 stopSelf();
127             }
128         }
