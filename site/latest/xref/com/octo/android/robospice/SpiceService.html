102             cacheManager = createCacheManager(getApplication());
103         } catch (CacheCreationException e) {
104             Ln.e(e);

114         final RequestListenerNotifier progressReporter = createRequestRequestListenerNotifier();
115         final SpiceServiceListenerNotifier spiceServiceListenerNotifier = createSpiceServiceListenerNotifier();
116         final RequestProcessorListener requestProcessorListener = createRequestProcessorListener();
117         final ExecutorService executorService = getExecutorService();
118         final NetworkStateChecker networkStateChecker = getNetworkStateChecker();
119         final RequestProgressManager requestProgressManager = createRequestProgressManager(requestProcessorListener, progressReporter, spiceServiceListenerNotifier);
120         final RequestRunner requestRunner = createRequestRunner(executorService, networkStateChecker, requestProgressManager);
121 
122         requestProcessor = createRequestProcessor(cacheManager, requestProgressManager, requestRunner);
123         requestProcessor.setFailOnCacheError(DEFAULT_FAIL_ON_CACHE_ERROR);
124 
125         notification = createDefaultNotification();
126 
127         Ln.d("SpiceService instance created.");
128     }

208     /**
209      * Factory method to create an {@link ExecutorService} that will be used to
210      * execute requests. The default implementation of this method will create a
211      * single threaded or multi-threaded {@link ExecutorService} depending on
212      * the number of threads returned by {@link #getThreadCount()} and will
213      * assign them the priority returned by {@link #getThreadPriority()}.
214      * <p/>
215      * If you override this method in your service, you can supply a custom
216      * {@link ExecutorService}.
217      * <p/>
218      * This feature has been implemented following a request from Riccardo
219      * Ciovati.
220      * @return the {@link ExecutorService} to be used to execute requests .
221      */
222     protected ExecutorService getExecutorService() {
223         final int threadCount = getThreadCount();
224         final int threadPriority = getThreadPriority();
225         if (threadCount <= 0) {
226             throw new IllegalArgumentException("Thread count must be >= 1");
227         } else {
228             return PriorityThreadPoolExecutor.getPriorityExecutor(threadCount, threadPriority);
229         }
230     }

251     @SuppressWarnings("deprecation")
252     public Notification createDefaultNotification() {
253 
254         Notification notification = null;
255         if (android.os.Build.VERSION.SDK_INT >= Build.VERSION_CODES.JELLY_BEAN) {
256             notification = new Notification.Builder(this).setSmallIcon(getApplicationInfo().icon).build();
257         } else if (android.os.Build.VERSION.SDK_INT >= Build.VERSION_CODES.HONEYCOMB) {
258             notification = new Notification.Builder(this).setSmallIcon(getApplicationInfo().icon).getNotification();
259         } else {
260             notification = new Notification();
261             notification.icon = getApplicationInfo().icon;
262             // temporary fix https://github.com/octo-online/robospice/issues/200
263             PendingIntent pendingIntent = PendingIntent.getActivity(getApplicationContext(), 0, new Intent(), 0);
264             notification.setLatestEventInfo(this, "", "", pendingIntent);
265             notification.tickerText = null;
266             notification.when = System.currentTimeMillis();
267         }
268 
269         if (android.os.Build.VERSION.SDK_INT >= Build.VERSION_CODES.JELLY_BEAN) {
270             notification.priority = Notification.PRIORITY_MIN;
271         }
272 
273         return notification;
274     }

430     private void stopIfNotBoundAndHasNoPendingRequests() {
431         Ln.v("Pending requests : " + currentPendingRequestCount);
432         if (currentPendingRequestCount == 0 && !isBound) {
433             stopSelf();
434         }

437     private void showNotificationIfNotBoundAndHasPendingRequestsOtherwiseHideNotification() {
438         // http://stackoverflow.com/a/13359680/693752
439         if (notification == null || isJUnit) {
440             return;
441         }
442         Ln.v("Pending requests : " + currentPendingRequestCount);
443         if (isBound || currentPendingRequestCount == 0) {
444             Ln.v("Stop foreground");
445             stopForeground(true);
446         } else {
447             Ln.v("Start foreground");
448             startForeground(notification);
449         }
450     }

452     private void startForeground(final Notification notification) {
453         try {
454             final Method setForegroundMethod = Service.class.getMethod("startForeground", int.class, Notification.class);
455             setForegroundMethod.invoke(this, getNotificationId(), notification);
456         } catch (final SecurityException e) {
457             Ln.e(e, "Unable to start a service in foreground");
458         } catch (final NoSuchMethodException e) {
459             Ln.e(e, "Unable to start a service in foreground");
460         } catch (final IllegalArgumentException e) {
461             Ln.e(e, "Unable to start a service in foreground");
462         } catch (final IllegalAccessException e) {
463             Ln.e(e, "Unable to start a service in foreground");
464         } catch (final InvocationTargetException e) {
465             Ln.e(e, "Unable to start a service in foreground");
466         }
467     }
