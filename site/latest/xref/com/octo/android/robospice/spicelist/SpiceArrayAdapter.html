179     protected void loadBitmapAsynchronously(T octo, ImageView thumbImageView,
180         String tempThumbnailImageFileName) {
181         if (thumbImageView.getTag() != null
182             && thumbImageView.getTag().equals(tempThumbnailImageFileName)) {
183             return;
184         }
185 
186         if (cancelPotentialWork(tempThumbnailImageFileName, thumbImageView)) {
187             final BitmapWorkerTask task = new BitmapWorkerTask(thumbImageView,
188                 octo);
189             task.fileName = tempThumbnailImageFileName;
190             final AsyncDrawable asyncDrawable = new AsyncDrawable(getContext()
191                 .getResources(), task);
192             thumbImageView.setImageDrawable(asyncDrawable);
193             thumbImageView.setTag(tempThumbnailImageFileName);
194             int width = Math.round(convertDpToPixel(IMG_WIDTH, getContext()));
195             int height = width;
196             task.execute(tempThumbnailImageFileName, String.valueOf(width),
197                 String.valueOf(height));
198         }
199     }

211     private boolean cancelPotentialWork(String fileName, ImageView imageView) {
212         final BitmapWorkerTask bitmapWorkerTask = getBitmapWorkerTask(imageView);
213 
214         if (bitmapWorkerTask != null) {
215             final String bitmapFileName = bitmapWorkerTask.fileName;
216             if (bitmapFileName == null || !bitmapFileName.equals(fileName)) {
217                 // Cancel previous task
218                 bitmapWorkerTask.cancel(true);
219             } else {
220                 // The same work is already in progress
221                 return false;
222             }
223         }
224         // No task associated with the ImageView, or an existing task was
225         // cancelled
226         return true;
227     }

230         if (imageView != null) {

241     private static Bitmap decodeSampledBitmapFromResource(String fileName,
242         int reqWidth, int reqHeight) {
243         // First decode with inJustDecodeBounds=true to check dimensions
244         final BitmapFactory.Options options = new BitmapFactory.Options();
245         options.inJustDecodeBounds = true;
246         BitmapFactory.decodeFile(fileName, options);
247 
248         // Calculate inSampleSize
249         options.inSampleSize = calculateInSampleSize(options, reqWidth,
250             reqHeight);
251 
252         // Decode bitmap with inSampleSize set
253         options.inJustDecodeBounds = false;
254         return BitmapFactory.decodeFile(fileName, options);
255     }
256 
257     private static int calculateInSampleSize(BitmapFactory.Options options,
258         int reqWidth, int reqHeight) {
259         // Raw height and width of image
260         final int height = options.outHeight;
261         final int width = options.outWidth;
262         int inSampleSize = 1;
263 
264         if (height > reqHeight || width > reqWidth) {
265             if (width > height) {
266                 inSampleSize = Math.round((float) height / (float) reqHeight);
267             } else {
268                 inSampleSize = Math.round((float) width / (float) reqWidth);
269             }
270         }
271         return inSampleSize;
272     }

310     protected class ThumbnailAsynTask extends AsyncTask<Object, Void, Boolean> {
311 
312         private T octo;
313         private SpiceListItemView<T> spiceListItemView;
314         private String tempThumbnailImageFileName = "";
315         private BigBinaryRequest request;
316 
317         public ThumbnailAsynTask(BigBinaryRequest request) {
318             this.request = request;
319         }
320 
321         @SuppressWarnings("unchecked")
322         @Override
323         protected Boolean doInBackground(Object... params) {
324             octo = (T) params[0];
325             spiceListItemView = (SpiceListItemView<T>) params[1];
326 
327             File tempThumbnailImageFile = request.getCacheFile();
328             tempThumbnailImageFileName = tempThumbnailImageFile
329                 .getAbsolutePath();
330             Ln.d("Filename : " + tempThumbnailImageFileName);
331 
332             if (!tempThumbnailImageFile.exists()) {
333                 if (isNetworkFetchingAllowed) {
334                     OctoImageRequestListener octoImageRequestListener = new OctoImageRequestListener(
335                         octo, spiceListItemView, tempThumbnailImageFileName);
336                     spiceManagerBinary.execute(request,
337                         "THUMB_IMAGE_" + octo.hashCode(),
338                         DurationInMillis.NEVER, octoImageRequestListener);
339                 }
340                 return false;
341             }
342             return true;
343         }
344 
345         @Override
346         protected void onPostExecute(Boolean isImageAvailableInCache) {
347             if (isImageAvailableInCache) {
348                 loadBitmapAsynchronously(octo,
349                     spiceListItemView.getImageView(),
350                     tempThumbnailImageFileName);
351             } else {
352                 spiceListItemView.getImageView().setImageDrawable(
353                     defaultDrawable);
354             }
355         }

358     private class BitmapWorkerTask extends AsyncTask<String, Void, Bitmap> {
359         private final WeakReference<ImageView> imageViewReference;
360         String fileName = "";
361         private T data;
362         private Animation animation;
363 
364         public BitmapWorkerTask(ImageView imageView, T data) {
365             // Use a WeakReference to ensure the ImageView can be garbage
366             // collected
367             imageViewReference = new WeakReference<ImageView>(imageView);
368             this.data = data;
369         }
370 
371         // Decode image in background.
372         @Override
373         protected Bitmap doInBackground(String... params) {
374             animation = AnimationUtils.loadAnimation(getContext(),
375                 android.R.anim.fade_in);
376             animation.setDuration(getContext().getResources().getInteger(
377                 android.R.integer.config_mediumAnimTime));
378             fileName = params[0];
379             int width = Integer.valueOf(params[1]);
380             int height = Integer.valueOf(params[2]);
381             return decodeSampledBitmapFromResource(fileName, width, height);
382         }
383 
384         // Once complete, see if ImageView is still around and set bitmap.
385         @Override
386         protected void onPostExecute(Bitmap bitmap) {
387             if (isCancelled()) {
388                 if (data.toString().equals("JFA")) {
389                     Ln.d(data.toString() + " : cancel decoding bitmap");
390                 }
391                 bitmap = null;
392             }
393 
394             if (imageViewReference != null && bitmap != null) {
395                 if (data.toString().equals("JFA")) {
396                     Ln.d(data.toString() + " : bitmapworkertask");
397                 }
398                 final ImageView imageView = imageViewReference.get();
399                 final BitmapWorkerTask bitmapWorkerTask = getBitmapWorkerTask(imageView);
400                 if (this == bitmapWorkerTask && imageView != null) {
401                     if (freshDrawableSet.contains(data)) {
402                         freshDrawableSet.remove(data);
403                         imageView.startAnimation(animation);
404                     }
405                     imageView.setImageBitmap(bitmap);
406                     // no used anymore.
407                     // imageView.setTag( null );
408                 }
409             }
410         }
411     }

453     /**
454      * This method convets dp unit to equivalent device specific value in
455      * pixels.
456      * @see http
457      *      ://stackoverflow.com/questions/4605527/converting-pixels-to-dp-in-
458      *      android
459      * @param dp
460      *            A value in dp(Device independent pixels) unit. Which we need
461      *            to convert into pixels
462      * @param context
463      *            Context to get resources and device specific display metrics
464      * @return A float value to represent Pixels equivalent to dp according to
465      *         device
466      */
467     public static float convertDpToPixel(float dp, Context context) {
468         Resources resources = context.getResources();
469         DisplayMetrics metrics = resources.getDisplayMetrics();
470         float px = dp * (metrics.densityDpi / DP_TO_PX_CONVERSION_FACTOR);
471         return px;
