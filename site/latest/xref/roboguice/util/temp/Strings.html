27      /**
28       * Like join, but allows for a distinct final delimiter. For english
29       * sentences such as "Alice, Bob and Charlie" use ", " and " and " as the
30       * delimiters.
31       * @param delimiter
32       *            usually ", "
33       * @param lastDelimiter
34       *            usually " and "
35       * @param objs
36       *            the objects
37       * @param <T>
38       *            the type
39       * @return a string
40       */
41      public static <T> String joinAnd(final String delimiter,
42          final String lastDelimiter, final Collection<T> objs) {
43          if (objs == null || objs.isEmpty()) {
44              return "";
45          }
46  
47          final Iterator<T> iter = objs.iterator();
48          final StringBuilder buffer = new StringBuilder(Strings.toString(iter
49              .next()));
50          int i = 1;
51          while (iter.hasNext()) {
52              final T obj = iter.next();
53              if (notEmpty(obj)) {
54                  buffer.append(++i == objs.size() ? lastDelimiter : delimiter)
55                      .append(Strings.toString(obj));
56              }
57          }
58          return buffer.toString();
59      }

66      public static <T> String join(final String delimiter,
67          final Collection<T> objs) {
68          if (objs == null || objs.isEmpty()) {
69              return "";
70          }
71  
72          final Iterator<T> iter = objs.iterator();
73          final StringBuilder buffer = new StringBuilder(Strings.toString(iter
74              .next()));
75  
76          while (iter.hasNext()) {
77              final T obj = iter.next();
78              if (notEmpty(obj)) {
79                  buffer.append(delimiter).append(Strings.toString(obj));
80              }
81          }
82          return buffer.toString();

106     public static long copyLarge(Reader input, Writer output) {
107         try {
108             char[] buffer = new char[DEFAULT_BUFFER_SIZE];
109             long count = 0;
110             int n;
111             while (-1 != (n = input.read(buffer))) {
112                 output.write(buffer, 0, n);
113                 count += n;
114             }
115             return count;
116         } catch (IOException e) {
117             throw new RuntimeException(e);
118         }

142     public static String md5(String s) {
143         // http://stackoverflow.com/questions/1057041/difference-between-java-and-php5-md5-hash
144         // http://code.google.com/p/roboguice/issues/detail?id=89
145         try {
146 
147             final byte[] hash = MessageDigest.getInstance("MD5").digest(
148                 s.getBytes(CharEncoding.UTF_8));
149             final StringBuilder hashString = new StringBuilder();
150 
151             for (byte aHash : hash) {
152                 String hex = Integer.toHexString(aHash);
153 
154                 if (hex.length() == 1) {
155                     hashString.append('0');
156                     hashString.append(hex.charAt(hex.length() - 1));
157                 } else {
158                     hashString.append(hex.substring(hex.length() - 2));
159                 }
160             }
161 
162             return hashString.toString();
163 
164         } catch (Exception e) {
165             throw new RuntimeException(e);
166         }
167     }

184     public static String[] chunk(String str, int chunkSize) {
185         if (isEmpty(str) || chunkSize == 0) {
186             return new String[0];
187         }
188 
189         final int len = str.length();
190         final int arrayLen = (len - 1) / chunkSize + 1;
191         final String[] array = new String[arrayLen];
192         for (int i = 0; i < arrayLen; ++i) {
193             array[i] = str.substring(i * chunkSize,
194                 i * chunkSize + chunkSize < len ? i * chunkSize + chunkSize
195                     : len);
196         }
197 
198         return array;
199     }

210     public static String namedFormat(String str, Object... nameValuePairs) {
211         if (nameValuePairs.length % 2 != 0) {
212             throw new InvalidParameterException(
213                 "You must include one value for each parameter");
214         }
215 
216         final HashMap<String, String> map = new HashMap<String, String>(
217             nameValuePairs.length / 2);
218         for (int i = 0; i < nameValuePairs.length; i += 2) {
219             map.put(Strings.toString(nameValuePairs[i]),
220                 Strings.toString(nameValuePairs[i + 1]));
221         }
222 
223         return namedFormat(str, map);
224     }
